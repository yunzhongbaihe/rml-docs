var rightTemplate = [
    {label: '粘贴', accelerator: 'ctrl+c'},
    {label: '复制', accelerator: 'ctrl+v'},
];
var rightMenu = Menu.buildFromTemplate(rightTemplate); // 构建右键菜单
window.addEventListener('contextmenu', (e) => {
    e.preventDefault();
    rightMenu.popup({window: remote.getCurrentWindow});
});
