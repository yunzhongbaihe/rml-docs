2、index.json

Page({
    data: {
        todos: ['1', '2', '3'],
        msg: 'Wechat Mini Program'
    },
    handleInput(e){
        this.setData({
            msg: e.detail.value,
        });
    },
    addTodo(){
        this.setData({
            msg: '',
            todos: [...this.data.todos, this.data.msg],
        });
    },
    takePhoto(){
        const ctx = wx.createCameraContext();
        ctx.takePhoto({
            quality: 'high',
            success: res => {
                this.setData({
                    src: res.tempImagePath,
                });
            }
        });
    }
});
